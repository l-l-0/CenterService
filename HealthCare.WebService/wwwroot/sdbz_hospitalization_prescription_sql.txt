SELECT DPM.PATIENT_ID,
       DPM.VISIT_ID,
       DPM.DISPENSARY,
       DPD.PRESC_DATE,
       DPD.PRESC_NO,
       DPD.ITEM_NO,
       DPD.ORDER_NO,
       DPD.ORDER_SUB_NO,
       DPD.DRUG_CODE,
       DPD.DRUG_SPEC,
       DPD.DRUG_NAME,
       DPD.FIRM_ID,
       DPD.PACKAGE_SPEC,
       DPD.PACKAGE_UNITS,
       DPD.QUANTITY,
       DPD.COSTS,
       DPD.PAYMENTS,
       DPM.PRESC_STATUS,
       DPD.ADMINISTRATION,
       DPD.DOSAGE,
       DPD.DOSAGE_UNITS,
       DPM.PRESC_TYPE,
       DPD.AMOUNT_PER_PACKAGE,
       DPD.FREQUENCY,
       DPD.DOSAGE_EACH,
       DPD.FREQ_DETAIL,
       DPM.DOCTOR_USER,
       DPD.PRESC_FLAG
       FROM DOCT_DRUG_PRESC_DETAIL DPD, DOCT_DRUG_PRESC_MASTER DPM
 WHERE DPD.PRESC_NO = DPM.PRESC_NO
   and DPD.PRESC_DATE = DPM.PRESC_DATE
   and DPM.COSTS >= 0
   AND DPM.PRESC_STATUS = 0
   and DPD.PRESC_FLAG is null
   AND DPM.PRESC_DATE >= TO_DATE('2018-09-01', 'YYYY-MM-DD')
   and dpm.dispensary in ('020710','020805')
