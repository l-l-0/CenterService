SELECT OOC.PATIENT_ID,
        OOC.VISIT_DATE,
        OOC.VISIT_NO,
        OOC.SERIAL_NO,
        OOC.ORDER_CLASS,
        OOC.ORDER_NO,
        OOC.ORDER_SUB_NO,
        OP.PRESC_NO,
        OOC.ITEM_NO,
        OP.DRUG_CODE,
        OP.DRUG_NAME,
        OP.DRUG_SPEC,
        OP.FIRM_ID,
        OP.UNITS,
        OP.AMOUNT,
        OP.COSTS,
        OP.DISPENSARY,
        OOC.ORDERED_BY_DEPT,
        OOC.ORDERED_BY_DOCTOR
FROM OUTP_PRESC OP,OUTP_ORDERS_COSTS OOC
WHERE OOC.VISIT_DATE = OP.VISIT_DATE
AND     OOC.VISIT_NO   =  OP.VISIT_NO
AND     OOC.SERIAL_NO = OP.SERIAL_NO
AND     OOC.ORDER_NO = OP.ORDER_NO
AND     OOC.ORDER_SUB_NO = OP.SUB_ORDER_NO
AND     OOC.ITEM_CLASS = 'A'
AND     OOC.CHARGE_INDICATOR = 0 
AND     OOC.VISIT_DATE >= TO_DATE('2018-09-03','YYYY-MM-DD')
and ooc.performed_by in ('020710','020805')
